class Solution {
    public int integerReplacement(int n) {
        return findMin(n);
    }

    private int findMin(int n) {
        if (n == 1) {
            return 0;
        }

        if (n % 2 == 0) {
            int even = findMin(n / 2);
        } else {
            int odd = Math.min(findMin(n - 1), findMin(n + 1));
        }

        return Math.min(even, odd);

    }
